#include <iostream>

double g1(double x, double y) {
	return sqrt((x * y + 5 * x - 1) / 2);
}

double g2(double x, double y) {
	return sqrt(x + 3 * log(x));
}

int main() {
	int cont = 0;
	double x = 4, y = 4, xant, yant;
	do {
		xant = x;
		yant = y;
		x = g1(x, y);
		y = g2(xant, y);
		cont++;
	} while (abs(x - xant) > 0.00001 && abs(y - yant) > 0.00001);

	std::cout << "x: " << x;
	std::cout << "\ny: " << y;
	std::cout << "\nn: " << cont;
	return 0;
}
